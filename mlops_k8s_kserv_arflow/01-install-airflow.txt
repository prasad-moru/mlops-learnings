# Phase 5 — Step 1: Install Airflow via Helm
# ─────────────────────────────────────────────────
# WHY LocalExecutor (not CeleryExecutor):
#   CeleryExecutor needs Redis + multiple worker pods = 6+ extra pods.
#   LocalExecutor runs tasks in the same pod as the scheduler.
#   Perfect for development. Upgrade to KubernetesExecutor in production.
#
# WHY separate namespace (airflow):
#   Airflow is infrastructure — not an ML service.
#   Keeps it isolated from mlops and monitoring namespaces.
#   Can be upgraded or removed without touching the ML stack.
# ─────────────────────────────────────────────────


# ── Add Apache Airflow Helm repo ──────────────────────────────────────────────
helm repo add apache-airflow https://airflow.apache.org
helm repo update


# Verify repo added
helm repo list
# Expected:
#   NAME            URL
#   apache-airflow  https://airflow.apache.org


# ── Create airflow namespace ──────────────────────────────────────────────────
kubectl create namespace airflow


# ── Install Airflow ───────────────────────────────────────────────────────────
# Uses airflow-values.yaml for all configuration
# Takes 3-5 minutes — Helm pulls images and waits for pods to be ready
helm install airflow apache-airflow/airflow \
  --namespace airflow \
  --version 1.11.0 \
  --values /home/devops/mlops-learnings/mlops_k8s_kserv_arflow/airflow-values.yaml \
  --set "webserver.defaultUser.enabled=true" \
  --set "webserver.defaultUser.role=Admin" \
  --set "webserver.defaultUser.username=admin" \
  --set "webserver.defaultUser.password=admin" \
  --set "webserver.defaultUser.email=admin@example.com" \
  --set "webserver.defaultUser.firstName=Admin" \
  --set "webserver.defaultUser.lastName=User" \
  --timeout 10m \
  --wait


helm upgrade airflow apache-airflow/airflow \
